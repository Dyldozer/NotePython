function main(workbook: ExcelScript.Workbook) {

  let sheet = workbook.getWorksheet("Sheet1");

  let input = sheet.getRange("B2").getValue();

  if (input === "" || input === null) {
    return;
  }

  let table = sheet.getTable("Table1");
  table.addRow(-1, [input]);

  sheet.getRange("B2").clear();

}
